<canvas id="game" width="480" height="640"></canvas>


<script>

const canvas = document.getElementById("game");

const ctx = canvas.getContext("2d");


// Paddle

let paddleWidth = 80;

let paddleHeight = 12;

let paddleX = (canvas.width - paddleWidth) / 2;

let paddleDX = 0;


// Ball

let ballX = canvas.width / 2;

let ballY = canvas.height - 50;

let ballDX = 3;

let ballDY = -3;

let ballRadius = 8;


// Bricks

const rows = 5, cols = 7;

const brickW = 60, brickH = 20;

const brickPadding = 10;

const brickTop = 60, brickLeft = 25;

let bricks = [];

for (let r = 0; r < rows; r++) {

  bricks[r] = [];

  for (let c = 0; c < cols; c++) {

    bricks[r][c] = { x: 0, y: 0, hit: false };

  }

}


document.addEventListener("mousemove", e => {

  const rect = canvas.getBoundingClientRect();

  paddleX = e.clientX - rect.left - paddleWidth / 2;

});


function drawBall() {

  ctx.beginPath();

  ctx.arc(ballX, ballY, ballRadius, 0, Math.PI * 2);

  ctx.fillStyle = "#ff82b2";

  ctx.fill();

}


function drawPaddle() {

  ctx.fillStyle = "#ffc9e6";

  ctx.fillRect(paddleX, canvas.height - paddleHeight - 10, paddleWidth, paddleHeight);

}


function drawBricks() {

  for (let r = 0; r < rows; r++) {

    for (let c = 0; c < cols; c++) {

      if (!bricks[r][c].hit) {

        const x = brickLeft + c * (brickW + brickPadding);

        const y = brickTop + r * (brickH + brickPadding);

        bricks[r][c].x = x;

        bricks[r][c].y = y;


        ctx.fillStyle = ["#ffd5ea","#ffdff5","#ffeafc","#ffccf1","#ffd1f7"][r];

        ctx.fillRect(x, y, brickW, brickH);

      }

    }

  }

}


function detectBrickCollision() {

  for (let r = 0; r < rows; r++) {

    for (let c = 0; c < cols; c++) {

      const b = bricks[r][c];

      if (!b.hit) {

        if (

          ballX > b.x && ballX < b.x + brickW &&

          ballY > b.y && ballY < b.y + brickH

        ) {

          ballDY = -ballDY;

          b.hit = true;

        }

      }

    }

  }

}


function update() {

  ctx.clearRect(0, 0, canvas.width, canvas.height);


  drawBall();

  drawPaddle();

  drawBricks();

  detectBrickCollision();


  ballX += ballDX;

  ballY += ballDY;


  // Wall bounce

  if (ballX + ballDX > canvas.width - ballRadius || ballX + ballDX < ballRadius) {

    ballDX = -ballDX;

  }

  if (ballY + ballDY < ballRadius) {

    ballDY = -ballDY;

  }


  // Paddle bounce

  if (

    ballY + ballRadius > canvas.height - paddleHeight - 10 &&

    ballX > paddleX &&

    ballX < paddleX + paddleWidth

  ) {

    ballDY = -ballDY;

  }


  // Game over

  if (ballY > canvas.height) {

    document.location.reload();

  }


  requestAnimationFrame(update);

}


update();

</script>
